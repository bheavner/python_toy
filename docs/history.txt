# -*- coding: utf-8 -*-   
"""A unit testing module for python_toy"""

import unittest

class ToyTests(unittest.TestCase):
    """Toy Unit Test."""

    def test_unittest(self):
        """Double check that unittest is working."""
        self.assertTrue(False)

if __name__ == '__main__':
    unittest.main()
